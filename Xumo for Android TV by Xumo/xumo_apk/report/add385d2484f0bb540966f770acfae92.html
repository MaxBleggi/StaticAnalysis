import android.net.Uri;

public final class AesCipherDataSource implements DataSource {
    @Nullable
    private AesFlushingCipher cipher;
    private final byte[] secretKey;
    private final DataSource upstream;

        return open;
    }

    @Nullable

        this.cipher = null;

}
