    public float getInterpolation(float f) {
        if (f <= 0.0f) {
            return 0.0f;
        }
        if (f >= 1.0f) {
            return 1.0f;
        }

        int length = this.mX.length - 1;
        while (length - i > 1) {
            int i2 = (i + length) / 2;

        float f3 = this.mY[i];
        return f3 + (f * (this.mY[length] - f3));
