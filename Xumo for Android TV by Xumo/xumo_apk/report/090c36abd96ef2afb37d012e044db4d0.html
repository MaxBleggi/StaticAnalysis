        public void onCheckedChanged(CompoundButton compoundButton, boolean z) {

                    ChipGroup.this.setCheckedId(compoundButton);
                } else if (ChipGroup.this.checkedId == compoundButton) {
                    ChipGroup.this.setCheckedId(-1);
                }
            }
        }
